057    package org.jfree.data.time;
058    
059    import java.io.Serializable;
060    import java.util.Calendar;
061    import java.util.Date;
062    import java.util.TimeZone;
063    
064    /**
065     * Represents a millisecond.  This class is immutable, which is a requirement 
066     * for all {@link RegularTimePeriod} subclasses.
067     */
068    public class Millisecond extends RegularTimePeriod implements Serializable {
069    
070        /** For serialization. */
071        static final long serialVersionUID = -5316836467277638485L;
072        
073        /** A constant for the first millisecond in a second. */
074        public static final int FIRST_MILLISECOND_IN_SECOND = 0;
075    
076        /** A constant for the last millisecond in a second. */
077        public static final int LAST_MILLISECOND_IN_SECOND = 999;
078    
079        /** The day. */
080        private Day day;
081        
082        /** The hour in the day. */
083        private byte hour;
084        
085        /** The minute. */
086        private byte minute;
087    
088        /** The second. */
089        private byte second;
090    
091        /** The millisecond. */
092        private int millisecond;
093    
094        /**
095         * The pegged millisecond. 
096         */
097        private long firstMillisecond;
098        
099        /**
100         * Constructs a millisecond based on the current system time.
101         */
102        public Millisecond() {
103            this(new Date());
104        }
105    
106        /**
107         * Constructs a millisecond.
108         *
109         * @param millisecond  the millisecond (0-999).
110         * @param second  the second.
111         */
112        public Millisecond(int millisecond, Second second) {
113            this.millisecond = millisecond;
114            this.second = (byte) second.getSecond();
115            this.minute = (byte) second.getMinute().getMinute();
116            this.hour = (byte) second.getMinute().getHourValue();
117            this.day = second.getMinute().getDay();
118            peg(Calendar.getInstance());
119        }
120    
121        /**
122         * Creates a new millisecond.
123         * 
124         * @param millisecond  the millisecond (0-999).
125         * @param second  the second (0-59).
126         * @param minute  the minute (0-59).
127         * @param hour  the hour (0-23).
128         * @param day  the day (1-31).
129         * @param month  the month (1-12).
130         * @param year  the year (1900-9999).
131         */    
132        public Millisecond(int millisecond, int second, int minute, int hour,
133                           int day, int month, int year) {
134                               
135            this(millisecond, new Second(second, minute, hour, day, month, year));
136        
137        }
138    
139        /**
140         * Constructs a millisecond.
141         *
142         * @param time  the time.
143         */
144        public Millisecond(Date time) {
145            this(time, RegularTimePeriod.DEFAULT_TIME_ZONE);
146        }
147    
148        /**
149         * Creates a millisecond.
150         *
151         * @param time  the instant in time.
152         * @param zone  the time zone.
153         */
154        public Millisecond(Date time, TimeZone zone) {
155            Calendar calendar = Calendar.getInstance(zone);
156            calendar.setTime(time);
157            this.millisecond = calendar.get(Calendar.MILLISECOND);
158            this.second = (byte) calendar.get(Calendar.SECOND);
159            this.minute = (byte) calendar.get(Calendar.MINUTE);
160            this.hour = (byte) calendar.get(Calendar.HOUR_OF_DAY);
161            this.day = new Day(time, zone);
162            peg(calendar);
163        }
164    
165        /**
166         * Returns the second.
167         *
168         * @return The second.
169         */
170        public Second getSecond() {
171            return new Second(this.second, this.minute, this.hour, 
172                    this.day.getDayOfMonth(), this.day.getMonth(), 
173                    this.day.getYear());
174        }
175    
176        /**
177         * Returns the millisecond.
178         *
179         * @return The millisecond.
180         */
181        public long getMillisecond() {
182            return this.millisecond;
183        }
184    
185        /**
186         * Returns the first millisecond of the second.  This will be determined 
187         * relative to the time zone specified in the constructor, or in the 
188         * calendar instance passed in the most recent call to the 
189         * {@link #peg(Calendar)} method.
190         *
191         * @return The first millisecond of the second.
192         * 
193         * @see #getLastMillisecond()
194         */
195        public long getFirstMillisecond() {
196            return this.firstMillisecond;
197        }
198    
199        /**
200         * Returns the last millisecond of the second.  This will be 
201         * determined relative to the time zone specified in the constructor, or
202         * in the calendar instance passed in the most recent call to the 
203         * {@link #peg(Calendar)} method.
204         *
205         * @return The last millisecond of the second.
206         * 
207         * @see #getFirstMillisecond()
208         */
209        public long getLastMillisecond() {
210            return this.firstMillisecond;
211        }
212        
213        /** 
214         * Recalculates the start date/time and end date/time for this time period 
215         * relative to the supplied calendar (which incorporates a time zone).
216         * 
217         * @param calendar  the calendar (<code>null</code> not permitted).
218         * 
219         * @since 1.0.3
220         */
221        public void peg(Calendar calendar) {
222            this.firstMillisecond = getFirstMillisecond(calendar);
223        }
224    
225        /**
226         * Returns the millisecond preceding this one.
227         *
228         * @return The millisecond preceding this one.
229         */
230        public RegularTimePeriod previous() {
231    
232            RegularTimePeriod result = null;
233    
234            if (this.millisecond != FIRST_MILLISECOND_IN_SECOND) {
235                result = new Millisecond(this.millisecond - 1, getSecond());
236            }
237            else {
238                Second previous = (Second) getSecond().previous();
239                if (previous != null) {
240                    result = new Millisecond(LAST_MILLISECOND_IN_SECOND, previous);
241                }
242            }
243            return result;
244    
245        }
246    
247        /**
248         * Returns the millisecond following this one.
249         *
250         * @return The millisecond following this one.
251         */
252        public RegularTimePeriod next() {
253    
254            RegularTimePeriod result = null;
255            if (this.millisecond != LAST_MILLISECOND_IN_SECOND) {
256                result = new Millisecond(this.millisecond + 1, getSecond());
257            }
258            else {
259                Second next = (Second) getSecond().next();
260                if (next != null) {
261                    result = new Millisecond(FIRST_MILLISECOND_IN_SECOND, next);
262                }
263            }
264            return result;
265    
266        }
267    
268        /**
269         * Returns a serial index number for the millisecond.
270         *
271         * @return The serial index number.
272         */
273        public long getSerialIndex() {
274            long hourIndex = this.day.getSerialIndex() * 24L + this.hour;
275            long minuteIndex = hourIndex * 60L + this.minute;
276            long secondIndex = minuteIndex * 60L + this.second;
277            return secondIndex * 1000L + this.millisecond;
278        }
279    
280        /**
281         * Tests the equality of this object against an arbitrary Object.
282         * <P>
283         * This method will return true ONLY if the object is a Millisecond object
284         * representing the same millisecond as this instance.
285         *
286         * @param obj  the object to compare
287         *
288         * @return <code>true</code> if milliseconds and seconds of this and object
289         *      are the same.
290         */
291        public boolean equals(Object obj) {
292            if (obj == this) {
293                return true;
294            }
295            if (!(obj instanceof Millisecond)) {
296                return false;
297            }
298            Millisecond that = (Millisecond) obj;
299            if (this.millisecond != that.millisecond) {
300                return false;
301            }
302            if (this.second != that.second) {
303                return false;
304            }
305            if (this.minute != that.minute) {
306                return false;
307            }
308            if (this.hour != that.hour) {
309                return false;
310            }
311            if (!this.day.equals(that.day)) {
312                return false;
313            }
314            return true;
315        }
316    
317        /**
318         * Returns a hash code for this object instance.  The approach described by 
319         * Joshua Bloch in "Effective Java" has been used here:
320         * <p>
321         * <code>http://developer.java.sun.com/developer/Books/effectivejava
322         * /Chapter3.pdf</code>
323         * 
324         * @return A hashcode.
325         */
326        public int hashCode() {
327            int result = 17;
328            result = 37 * result + this.millisecond;
329            result = 37 * result + getSecond().hashCode();
330            return result;
331        }
332    
333        /**
334         * Returns an integer indicating the order of this Millisecond object
335         * relative to the specified object:
336         *
337         * negative == before, zero == same, positive == after.
338         *
339         * @param obj  the object to compare
340         *
341         * @return negative == before, zero == same, positive == after.
342         */
343        public int compareTo(Object obj) {
344    
345            int result;
346            long difference;
347    
348            // CASE 1 : Comparing to another Second object
349            // -------------------------------------------
350            if (obj instanceof Millisecond) {
351                Millisecond ms = (Millisecond) obj;
352                difference = getFirstMillisecond() - ms.getFirstMillisecond();
353                if (difference > 0) {
354                    result = 1;
355                }
356                else {
357                    if (difference < 0) {
358                        result = -1;
359                    }
360                    else {
361                        result = 0;
362                    }
363                }
364            }
365    
366            // CASE 2 : Comparing to another TimePeriod object
367            // -----------------------------------------------
368            else if (obj instanceof RegularTimePeriod) {
369                // more difficult case - evaluate later...
370                result = 0;
371            }
372    
373            // CASE 3 : Comparing to a non-TimePeriod object
374            // ---------------------------------------------
375            else {
376                // consider time periods to be ordered after general objects
377                result = 1;
378            }
379    
380            return result;
381    
382        }
383    
384        /**
385         * Returns the first millisecond of the time period.
386         *
387         * @param calendar  the calendar (<code>null</code> not permitted).
388         *
389         * @return The first millisecond of the time period.
390         *
391         * @throws NullPointerException if <code>calendar</code> is 
392         *     <code>null</code>.
393         */
394        public long getFirstMillisecond(Calendar calendar) {
395            int year = this.day.getYear();
396            int month = this.day.getMonth() - 1;
397            int day = this.day.getDayOfMonth();
398            calendar.clear();
399            calendar.set(year, month, day, this.hour, this.minute, this.second);
400            calendar.set(Calendar.MILLISECOND, this.millisecond);
401            //return calendar.getTimeInMillis();  // this won't work for JDK 1.3
402            return calendar.getTime().getTime();
403        }
404    
405        /**
406         * Returns the last millisecond of the time period.
407         *
408         * @param calendar  the calendar (<code>null</code> not permitted).
409         *
410         * @return The last millisecond of the time period.
411         *
412         * @throws NullPointerException if <code>calendar</code> is 
413         *     <code>null</code>.
414         */
415        public long getLastMillisecond(Calendar calendar) {
416            return getFirstMillisecond(calendar);
417        }
